# 1. 两数之和(Two Sum)

* [描述和解法说明](https://www.yuque.com/tokido/xquvcp/pnbs4r)
  
### 双循环解法:
```js
var twoSum = function(nums, target) {
    for(let i=0; i<nums.length; i++) {
        for(let j=i+1; j<nums.length; j++) {
            if(nums[i] + nums[j] === target) return [i, j];
        }
    }
};
```

### 哈希表解法
```js
var twoSum = function(nums, target) {
    let map = {};
    for(let i=0; i<nums.length; i++) {
        if(map[nums[i]] === true) return [map[target-nums[i]], i];
        if(Array.isArray(map[nums[i]])) {
            map[nums[i]].push(i);
            return map[nums[i]];
        }
        if(map[nums[i]] === undefined) {
            // 对输入为重复元素时的处理, 如输入nums=[3, 3], target=6时
            if(target === 2 * nums[i]) {
                map[nums[i]] = [i]
                continue;
            }
            map[nums[i]] = i;
            map[target - nums[i]] = true;
        }
    }
};
```
